### SEO Metadata

```typescript
// apps/web/app/(main)/projects/[slug]/[tab]/page.tsx
export async function generateMetadata({ params }): Promise<Metadata> {
  const project = getProject(params.slug);
  const tab = project?.tabs.find(t => t.id === params.tab);
  return {
    title: `${tab?.label} | ${project?.title}`,
    description: project?.tagline,
    openGraph: {
      images: [`/api/og/projects/${params.slug}`],
    },
  };
}
```
