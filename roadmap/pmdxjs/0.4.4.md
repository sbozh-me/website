## Page Implementation

```typescript
// app/(main)/cv/page.tsx
import { compile } from '@sbozh/pmdxjs/runtime/server';
import cvContent from '@/content/cv.pmdx';
import { CVPrintButton } from '@/components/cv/CVPrintButton';

export default async function CVPage() {
  const CVComponent = await compile(cvContent);

  return (
    <div className="mx-auto px-6 md:px-12 lg:px-24 py-12">
      <div className="flex justify-end mb-4 print:hidden">
        <CVPrintButton />
      </div>
      <CVComponent />
    </div>
  );
}
```
